struct sbgc_frame_t {
  bool ok;
  uint8_t cmd_id;
  uint8_t data_size;
  uint8_t header_checksum;
  uint8_t * data;
  uint8_t data_checksum;

  sbgc_frame_t();
  ~sbgc_frame_t();
};

SBGC UART protocol frame

struct sbgc_realtime_data_t {
  vec3_t accel;
  vec3_t gyro;
  vec3_t camera_angles;
  vec3_t frame_angles;
  vec3_t rc_angles;
  vec3_t encoder_angles;
  int cycle_time;
  int i2c_error_count;
  int system_error;
  int battery_level;

  sbgc_realtime_data_t();
  ~sbgc_realtime_data_t();
};

SBGC realtime data

struct sbgc_t {
  bool connected;
  sbgc_realtime_data_t data;
  std::string port;
  int serial;
  uint8_t board_version;
  uint16_t firmware_version;
  uint8_t debug_mode;
  uint16_t board_features;
  uint8_t connection_flags;

  sbgc_t();
  sbgc_t(const std::string &port_);
};
